${escapeHtml(p.title)}

${escapeHtml(p.description)}

Teknologi: ${escapeHtml(p.tech.join(', '))}

Lihat demo

`).join(''); } }catch(err){ projectsList.innerHTML = 'Gagal memuat proyek.'; console.error(err); } } // kirim form kontak ke API (POST /api/contact) contactForm.addEventListener('submit', async (e)=>{ e.preventDefault(); const fd = new FormData(contactForm); const payload = Object.fromEntries(fd.entries()); contactResult.textContent = 'Mengirim...'; try{ const res = await fetch('/api/contact',{ method:'POST', headers:{ 'Content-Type': 'application/json' }, body: JSON.stringify(payload) }); const data = await res.json(); if(res.ok){ contactResult.textContent = data.message || 'Pesan terkirim!'; contactForm.reset(); }else{ contactResult.textContent = data.error || 'Gagal mengirim pesan'; } }catch(err){ contactResult.textContent = 'Gagal mengirim — cek koneksi'; console.error(err); } }); // helper function escapeHtml(s){ if(!s) return ''; return s.replace(/[&<>"']/g, c => ({'&':'&','<':'<','>':'>','"':'"',"'":"'"})[c]); } // inisialisasi loadProjects(); ``` --- ## 4) `server.js` — contoh server Node.js + Express (REST API + serve static files) ```javascript // server.js const express = require('express'); const path = require('path'); const cors = require('cors'); const app = express(); app.use(cors()); app.use(express.json()); // data contoh (bisa diganti dengan DB nyata) const projects = [ { id:1, title:'Website Toko', description:'Toko online sederhana', tech:['HTML','CSS','JS','Node'], url:'https://contoh.com'}, { id:2, title:'Aplikasi Mobile', description:'Prototype aplikasi mobile', tech:['React Native'], url:'#'} ]; // ROUTES — REST API app.get('/api/projects', (req,res)=>{ // contoh endpoint RESTful: GET semua proyek res.json(projects); }); app.get('/api/projects/:id',(req,res)=>{ const p = projects.find(x=>x.id===Number(req.params.id)); if(!p) return res.status(404).json({error:'Project tidak ditemukan'}); res.json(p); }); app.post('/api/contact',(req,res)=>{ const {name,email,message} = req.body || {}; if(!name||!email||!message) return res.status(400).json({error:'Lengkapi nama, email dan pesan'}); // Di sini Anda bisa menyimpan ke DB, kirim email, dsb. console.log('Pesan kontak:', {name,email,message}); res.json({message: 'Terima kasih! Pesan Anda telah diterima.'}); }); // serve static files dari /public app.use(express.static(path.join(__dirname,'public'))); // fallback untuk SPA / index.html app.get('*', (req,res)=>{ res.sendFile(path.join(__dirname,'public','index.html')); }); const PORT = process.env.PORT || 3000; app.listen(PORT, ()=> console.log(`Server berjalan di port ${PORT}`)); ``` --- ## 5) `package.json` minimal ```json { "name": "portfolio-project", "version": "1.0.0", "main": "server.js", "scripts": { "start": "node server.js" }, "dependencies": { "cors": "^2.8.5", "express": "^4.18.2" } } ``` --- ## 6) Cara menjalankan lokal 1. Pastikan Node.js terpasang (versi LTS direkomendasikan). 2. Buka terminal di folder `portfolio-project`. 3. `npm install` 4. `node server.js` atau `npm start` 5. Buka `http://localhost:3000` di browser — front-end akan memanggil API pada jalur `/api/*`. --- ## 7) Contoh panggilan API (curl) - GET proyek: ``` curl http://localhost:3000/api/projects ``` - POST kontak: ``` curl -X POST http://localhost:3000/api/contact -H 'Content-Type: application/json' -d '{"name":"Ariq","email":"ariq@example.com","message":"Halo"}' ``` --- ## 8) Membuat server dapat diakses publik — beberapa opsi ### Opsi A — cepat (demo) : ngrok - Pasang `ngrok` lalu jalankan `ngrok http 3000`. Anda akan mendapat URL publik `https://xxxx.ngrok.io` yang meneruskan ke `localhost:3000`. Cocok untuk demo singkat. - Catatan: URL ngrok bersifat sementara (session-based) kecuali langganan berbayar. ### Opsi B — hosting gratis untuk front-end + backend gratis (disarankan split) - Front-end statis (`public/`) → GitHub Pages atau Cloudflare Pages (gratis). Jika menggunakan GitHub Pages, hanya bisa untuk static site (tanpa backend Node). - Backend Node → gunakan layanan seperti Render (free), Railway (free tier), Fly.io (free tier), atau Vercel (untuk serverless functions). Caranya: push repo ke GitHub dan koneksikan ke layanan, atur start command `npm start`. ### Opsi C — VPS (lebih fleksibel) - Sewa VPS murah (contoh: DigitalOcean, Vultr, atau penyedia lokal). Pada VPS: clone repo, `npm install`, jalankan `node server.js` di background (pm2 atau systemd). Arahkan domain/subdomain ke IP VPS via DNS A record. ### Opsi D — All-in-one (Render / Railway) - Render: buat "Web Service", sambungkan repo, build command `npm install`, start `npm start` — Render memberikan domain `your-service.onrender.com`. > Kalau Anda mau saya bisa tulis langkah-langkah spesifik untuk salah satu platform (contoh: deploy ke Render atau ke GitHub Pages + Render untuk backend). --- ## 9) Keamanan & catatan produksi singkat - Jangan simpan kredensial langsung di kode; gunakan environment variables. - Untuk form kontak produksi gunakan validasi sisi server, proteksi rate-limit, dan anti-spam (reCAPTCHA jika perlu). - Untuk file upload gunakan batas ukuran dan pengecekan tipe file. --- ## 10) Penutup — Checklist yang sudah tersedia - [x] Front-end HTML5 + CSS + JS (panggilan REST API) - [x] Server Node.js + Express menyajikan API `GET /api/projects` dan `POST /api/contact` - [x] Instruksi menjalankan lokal - [x] Opsi publikasi (ngrok, Render, VPS, GitHub Pages) --- Jika mau, saya bisa: - Membuat repo GitHub lengkap dengan file-file di atas (saya buatkan konten siap-push). - Menuliskan langkah langkah deploy 1-klik pada Render atau Railway secara lengkap. - Menambahkan database sederhana (SQLite / MongoDB Atlas) untuk menyimpan pesan kontak dan proyek.
